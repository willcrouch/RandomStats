# Moving N Day Average
mA <- function(x, n){rollapplyr(lag(x,1), width = n, FUN = mean, fill = NA)}
mHM = function(x, n) { n /rollapplyr(lag(x,1)^-1, width = n, FUN = sum, fill = NA) }
mGM = function(x, n) { exp(rollapplyr(log(lag(x,1)), width = n, FUN = prod, fill = NA)^(1/n)) }
# Moving N Day Variance
mVar <- function(x, n){rollapplyr(lag(x,1), width = n, FUN = var, fill = NA)}
# Moving Stochastic Oscillator
mSO <- function(x){(x - rollapplyr(lag(x,1), width = 14, FUN = min, fill = NA)) / 
                  (rollapplyr(lag(x,1), width = 14, FUN = max, fill = NA) - rollapplyr(lag(x,1), width = 14, FUN = min, fill = NA)) * 100}
# Spread between MA
mSpread = function(x, n1, n2){rollapplyr(lag(x,1), width = n1, FUN = mean, fill = NA) - rollapplyr(lag(x,1), width = n2, FUN = mean, fill = NA)}
# Spread between High and Low prior day
mHLSpread = function(x1, x2){lag(rollapplyr(x1, width = 1, FUN = max, fill = NA) - rollapplyr(x2, width = 1, FUN = min, fill = NA),1)}
# Spread between prior N days
mHLC = function(x, n){rollapplyr(lag(x,1), width = n, FUN = max, fill = NA) - rollapplyr(lag(x,1), width = n, FUN = min, fill = NA)}

Train = StockDF[250:2000,6:42]
Test = StockDF[2001:nrow(StockDF),6:42]

TrainX = data.matrix(Train[,2:ncol(Train)])
TrainX = array(TrainX, dim = c(nrow(TrainX), ncol(TrainX), 1))
TrainY = data.matrix(Train$adjClose)

TestX = data.matrix(Test[,2:ncol(Test)])
TestX = array(TestX, dim = c(nrow(TestX), ncol(TestX), 1))
TestY = data.matrix(Test$adjClose)


# Create Model
model = keras_model_sequential() %>%   
  layer_lstm(units=512, input_shape=c(ncol(TrainX), 1), activation="tanh", return_sequences = TRUE) %>%  
  layer_dropout(.5) %>%
  layer_lstm(units=512, activation = "tanh") %>%
  layer_dropout(.5) %>%
  layer_dense(units=512, activation = "tanh") %>%
  layer_dropout(.5) %>%
  layer_dense(units=512, activation = "tanh") %>%
  layer_dropout(.5) %>%
  layer_dense(units=1, activation = "linear")

# Define loss and optimization
model %>% compile(loss = 'mse'
                  ,optimizer = optimizer_rmsprop()
                  ,metrics = list("mean_squared_error")
)

# Fit model
history <- model %>% 
  fit(TrainX,
      TrainY, 
      #validation_split = .2,
      validation_data = list(TestX, TestY),
      batch_size = 32, 
      epochs = 100,
      callbacks = list(
        callback_early_stopping(patience = 20, 
                                restore_best_weights = FALSE),
        callback_reduce_lr_on_plateau(factor = 0.2, 
                                      patience = 5)
    )
    #)
      ,verbose = FALSE)

# Plot training and validation by epoch
plot(history)
